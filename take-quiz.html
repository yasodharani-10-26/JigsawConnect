import { getDatabase, ref, set } from "https://www.gstatic.com/firebasejs/10.14.0/firebase-database.js"; function saveQuizScoreToProfile(quizId, quizTitle, finalScore, totalQuestions) { const db = getDatabase(); // 1. localStorage నుండి లాగిన్ ఐడెంటిటీ తెచ్చుకోవడం const hrId = localStorage.getItem('studentHrId'); if (!hrId) { console.error("Student not logged in. Score not saved on cloud."); return; } const safeKey = hrId.replace(/[.#$\[\]\/]/g, '_'); // 2. Firebase లో సదరు స్టూడెంట్ ప్రొఫైల్ లో 'quizScores' నోడ్ కింద సేవ్ చేయడం const studentQuizRef = ref(db, `students/${safeKey}/quizScores/${quizId}`); set(studentQuizRef, { quizTitle: quizTitle || "Untitled Quiz", score: finalScore, total: totalQuestions, percentage: ((finalScore / totalQuestions) * 100).toFixed(1) + "%", timestamp: Date.now() }).then(() => { console.log("Score synced to student profile successfully!"); }).catch((err) => { console.error("Error saving score:", err); }); }a
