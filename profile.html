import { getDatabase, ref, get } from "https://www.gstatic.com/firebasejs/10.14.0/firebase-database.js"; const hrId = localStorage.getItem('studentHrId'); if (hrId) { const safeKey = hrId.replace(/[.#$\[\]\/]/g, '_'); const db = getDatabase(); // ప్రొఫైల్ మరియు క్విజ్ స్కోర్ల డేటా పొందడం get(ref(db, `students/${safeKey}`)).then((snapshot) => { if (snapshot.exists()) { const studentData = snapshot.val(); // స్టూడెంట్ బేసిక్ డీటెయిల్స్ UI కి బైండ్ చేయండి document.getElementById('profileName').innerText = studentData.name; document.getElementById('profileRoll').innerText = studentData.rollNo; // క్విజ్ హిస్టరీ/స్కోర్లను లూప్ లో డిస్‌ప్లే చేయడం const scoresContainer = document.getElementById('quizScoresList'); scoresContainer.innerHTML = ""; if (studentData.quizScores) { Object.keys(studentData.quizScores).forEach(qId => { const record = studentData.quizScores[qId]; scoresContainer.innerHTML += `
${record.quizTitle}: ${record.score}/${record.total} (${record.percentage})
`; }); } else { scoresContainer.innerHTML = "

No quizzes completed yet.

"; } } }); }
